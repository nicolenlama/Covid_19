R0<-function(x,inc){
  #esta funcion recibe los datos y el numero de dias
  # que se usan de incremento para hacer el ratio de la diferencia
  #entre los contagiados
  valor<-0
  for (i in inc:length(x)){
    valor[i]<-x[i]/x[i-inc+1]
  }
  return(valor)
}
limpiar<-function(matriz, i1,i2, j1,j2,x){
  for (i in i1:i2){
    for (j in j1:j2){
      if (is.na(matriz[i,j]) | is.nan(matriz[i,j])
          |is.infinite(matriz[i,j])) {matriz[i,j]<-x}
      
    }
  }
  return(matriz)
}
